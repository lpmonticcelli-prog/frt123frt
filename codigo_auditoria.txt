<?php

namespace App\Http\Controllers\Api\V1\Embarcador;

use App\Http\Controllers\Controller;

use App\Models\Carga;
use App\Http\Requests\StoreCargaRequest;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;
use Illuminate\Support\Facades\Storage;
use App\Jobs\ProcessarPublicacaoCarga;

class CargaController extends Controller
{
    /**
     * Publica uma nova carga no sistema (VIA FILA ASSÍNCRONA)
     */
    public function store(StoreCargaRequest $request)
    {
        $validated = $request->validated();
        $user = $request->user();

        // --- TRAVA DE SEGURANÇA (KYC EMBARCADOR) ---
        if ($user->status !== 'active') {
            return response()->json([
                'message' => 'Ação bloqueada. Sua conta não está ativa. Por favor, envie sua documentação empresarial no menu "Minha Conta" e aguarde a auditoria para publicar fretes.'
            ], 403);
        }

        // DISPARO ASSÍNCRONO PARA O REDIS (AGORA COMO SYNC NO .ENV)
        ProcessarPublicacaoCarga::dispatch(
            $validated,
            $user->embarcador->id,
            $request->ip(),
            $request->header('User-Agent')
        )->onQueue('default');

        return response()->json([
            'message' => 'Lote recebido com sucesso. Sua carga está sendo processada e aparecerá no mural em instantes.',
        ], 202);
    }

    /**
     * Lista as cargas disponíveis (Mural de Fretes) ou cargas do Embarcador
     */
    public function index(Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        // 1. Se for EMBARCADOR: Vê apenas os fretes da sua própria indústria (todos os status)
        if ($user->role && $user->role->slug === 'embarcador') {
            if (!$user->embarcador) {
                return response()->json([], 200);
            }
            $cargas = Carga::with(['motorista.user:id,name,email', 'publicacao_log', 'aceite_log'])
                ->where('embarcador_id', $user->embarcador->id)
                ->orderBy('created_at', 'desc')
                ->paginate(15);
        } 
        
        // 2. Se for MOTORISTA: Vê o MURAL DE FRETES (Apenas cargas "publicada" e sem dono)
        elseif ($user->role && $user->role->slug === 'motorista') {
            $cargas = Carga::with('embarcador:id,razao_social,cnpj')
                ->where('status', 'publicada')
                ->whereNull('motorista_id')
                ->orderBy('data_coleta', 'asc') // Mostra os mais urgentes primeiro
                ->paginate(15);
        } 
        
        // 3. Se for ADMIN/STAFF: Vê tudo
        else {
            $cargas = Carga::with(['embarcador:id,razao_social', 'motorista.user:id,name'])
                ->orderBy('created_at', 'desc')
                ->paginate(15);
        }

        return response()->json($cargas);
    }

    /**
     * Retorna os dados detalhados de uma carga específica
     */
    public function show(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para visualizar esta carga.'], 403);
        }

        $carga->load(['embarcador', 'motorista.user', 'aceite_log', 'publicacao_log']);

        return response()->json($carga);
    }

    /**
     * Atualiza os dados de uma carga existente
     */
    public function update(StoreCargaRequest $request, Carga $carga)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para alterar esta carga.'], 403);
        }

        if ($carga->status !== 'publicada') {
            return response()->json(['message' => 'Cargas em negociação ou já alocadas não podem ser editadas. Cancele-a se necessário.'], 403);
        }

        $validated = $request->validated();
        $taxaPlataforma = $validated['valor_frete'] * 0.05;

        $carga->update([
            'produto' => $validated['produto'],
            'especie' => $validated['especie'],
            'peso_kg' => $validated['peso_kg'],
            'cubagem_m3' => $validated['cubagem_m3'] ?? null,
            'tipo_veiculo' => $validated['tipo_veiculo'],
            'tipo_carroceria' => $validated['tipo_carroceria'],
            'cidade_origem' => $validated['cidade_origem'],
            'uf_origem' => strtoupper($validated['uf_origem']),
            'cidade_destino' => $validated['cidade_destino'],
            'uf_destino' => strtoupper($validated['uf_destino']),
            'distancia_km' => $validated['distancia_km'] ?? null,
            'valor_frete' => $validated['valor_frete'],
            'taxa_plataforma' => $taxaPlataforma,
            'data_coleta' => $validated['data_coleta'],
            'data_entrega_prevista' => $validated['data_entrega_prevista'] ?? null,
        ]);

        return response()->json([
            'message' => 'Carga atualizada com sucesso.',
            'carga' => $carga
        ]);
    }

    /**
     * Cancela (Exclusão Lógica) uma carga.
     */
    public function destroy(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para alterar esta carga.'], 403);
        }

        if ($carga->status !== 'publicada') {
            return response()->json(['message' => 'Esta carga não pode mais ser cancelada pois já está em negociação ou alocada.'], 403);
        }

        $carga->delete();

        return response()->json(['message' => 'Carga cancelada e removida do mural com sucesso.']);
    }

    /**
     * Motorista aceita a carga (MATCH SÍNCRONO DE ALTA PERFORMANCE)
     */
    public function aceitar(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'motorista');

        if (!$user->role || $user->role->slug !== 'motorista') {
            return response()->json(['message' => 'Apenas motoristas cadastrados podem aceitar fretes.'], 403);
        }

        if ($user->status !== 'active') {
            return response()->json([
                'message' => 'Ação bloqueada. Sua conta não está ativa. Envie sua CNH, selfie e documentos no menu "Minha Conta" para poder aceitar fretes na plataforma.'
            ], 403);
        }

        try {
            DB::transaction(function () use ($carga, $user, $request) {
                // TRAVA PESSIMISTA: Congela a linha para evitar Race Condition (Double Booking)
                $cargaLock = Carga::where('id', $carga->id)->lockForUpdate()->first();

                if ($cargaLock->status !== 'publicada') {
                    abort(409, 'Lamentamos, mas esta carga já foi aceita por outro motorista ou cancelada.');
                }

                $motoristaPerfilId = $user->motorista->id;

                // Geração do Contrato Imutável
                $termoContrato = "CONTRATO DE TRANSPORTE. O motorista {$user->name} aceita o frete ID {$cargaLock->id}, "
                               . "de {$cargaLock->cidade_origem} para {$cargaLock->cidade_destino}, "
                               . "pelo valor de R$ " . number_format($cargaLock->valor_frete, 2, ',', '.') . ".";
                
                $termoHash = hash('sha256', $termoContrato);

                // Efetiva o Match
                $cargaLock->update([
                    'status' => 'alocada',
                    'motorista_id' => $motoristaPerfilId
                ]);

                // Grava a auditoria
                DB::table('carga_aceites_log')->insert([
                    'carga_id' => $cargaLock->id,
                    'motorista_id' => $motoristaPerfilId,
                    'ip_address' => $request->ip(),
                    'user_agent' => $request->header('User-Agent'),
                    'termo_hash' => $termoHash,
                    'aceito_em' => now(),
                    'created_at' => now(),
                    'updated_at' => now(),
                ]);
            });

            return response()->json([
                'message' => 'Frete assinado e aceito com sucesso! Acesse Meus Fretes para gerenciar a viagem.',
            ], 200);

        } catch (\Symfony\Component\HttpKernel\Exception\HttpException $e) {
            return response()->json(['message' => $e->getMessage()], $e->getStatusCode());
        } catch (\Exception $e) {
            Log::error('Erro crítico no aceite do frete: ' . $e->getMessage());
            return response()->json(['message' => 'Ocorreu um erro interno ao processar sua assinatura.'], 500);
        }
    }

    /**
     * Retorna as cargas que o motorista logado aceitou
     */
    public function minhasCargas(Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'motorista');

        if (!$user->role || $user->role->slug !== 'motorista') {
            return response()->json(['message' => 'Acesso negado'], 403);
        }

        $cargas = Carga::with(['embarcador:id,razao_social', 'aceite_log', 'publicacao_log'])
            ->where('motorista_id', $user->motorista->id)
            ->latest()
            ->paginate(15);

        return response()->json($cargas);
    }

    /**
     * Motorista desiste da carga (Cancela o aceite)
     */
    public function cancelarAceite(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('motorista');

        if ($carga->motorista_id !== $user->motorista->id) {
            return response()->json(['message' => 'Acesso negado. Você não é o motorista desta carga.'], 403);
        }

        if ($carga->status !== 'alocada') {
            return response()->json(['message' => 'Esta carga já está em trânsito ou foi finalizada e não pode ser cancelada.'], 403);
        }

        $carga->update([
            'status' => 'publicada',
            'motorista_id' => null
        ]);

        return response()->json(['message' => 'Aceite cancelado. A carga retornou para o mural.']);
    }

    /**
     * Motorista informa que iniciou a viagem
     */
    public function iniciarViagem(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('motorista');

        if ($carga->motorista_id !== $user->motorista->id) {
            return response()->json(['message' => 'Acesso negado.'], 403);
        }

        if ($carga->status !== 'alocada') {
            return response()->json(['message' => 'Ação inválida para o status atual da carga.'], 400);
        }

        $carga->update(['status' => 'em_transito']);

        return response()->json(['message' => 'Viagem iniciada! Dirija com segurança.', 'carga' => $carga]);
    }

    /**
     * Motorista finaliza a entrega ENVIANDO AS FOTOS (POD)
     */
    public function finalizarEntrega(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('motorista');

        if ($carga->motorista_id !== $user->motorista->id) {
            return response()->json(['message' => 'Acesso negado.'], 403);
        }

        if ($carga->status !== 'em_transito') {
            return response()->json(['message' => 'A a carga precisa estar em trânsito para ser finalizada.'], 400);
        }

        $request->validate([
            'foto_canhoto' => 'required|file|mimes:jpeg,png,jpg,pdf|max:10240',
            'foto_carga'   => 'required|file|mimes:jpeg,png,jpg,pdf|max:10240',
        ], [
            'foto_canhoto.required' => 'Você precisa enviar a foto do canhoto assinado.',
            'foto_carga.required'   => 'Você precisa enviar a foto da carga entregue no destino.',
            'mimes'                 => 'Arquivo não permitido. Apenas imagens (JPEG, PNG) e PDFs são aceitos.'
        ]);

        $pathCanhoto = $request->file('foto_canhoto')->store('pod/' . $carga->id, 'public');
        $pathCarga = $request->file('foto_carga')->store('pod/' . $carga->id, 'public');

        $carga->update([
            'status' => 'entregue',
            'foto_canhoto' => $pathCanhoto,
            'foto_carga' => $pathCarga
        ]);

        return response()->json([
            'message' => 'Entrega concluída! Documentos recebidos com sucesso e em fase de auditoria.', 
            'carga' => $carga
        ]);
    }
}

<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\SoftDeletes;

class Carga extends Model
{
    use HasFactory, SoftDeletes;

    protected $fillable = [
        'embarcador_id',
        'motorista_id',
        'produto',
        'especie',
        'peso_kg',           
        'cubagem_m3',        
        'tipo_veiculo',      
        'tipo_carroceria',   
        'uf_origem',         
        'cidade_origem',     
        'uf_destino',        
        'cidade_destino',    
        'distancia_km',      
        'valor_frete',
        'taxa_plataforma',   
        'status',
        'foto_canhoto',      
        'foto_carga',        
        'data_coleta',       
        'data_entrega_prevista',
    ];

    protected $casts = [
        'data_coleta' => 'date',
        'data_entrega_prevista' => 'datetime',
        'peso_kg' => 'decimal:2',
        'cubagem_m3' => 'decimal:2',
        'distancia_km' => 'decimal:2',
        'valor_frete' => 'decimal:2',
        'taxa_plataforma' => 'decimal:2',
    ];

    public function embarcador()
    {
        return $this->belongsTo(Embarcador::class);
    }

    public function motorista()
    {
        return $this->belongsTo(Motorista::class);
    }

    /**
     * Relacionamento com a Auditoria Jurídica (Recibo Eletrônico do Motorista)
     * Renomeado para snake_case para dar "Match" perfeito com o Frontend Vue.js
     */
    public function aceite_log()
    {
        return $this->hasOne(CargaAceiteLog::class, 'carga_id', 'id');
    }

    /**
     * Relacionamento com a Auditoria Jurídica (Recibo Eletrônico do Embarcador)
     * Renomeado para snake_case para dar "Match" perfeito com o Frontend Vue.js
     */
    public function publicacao_log()
    {
        return $this->hasOne(CargaPublicacaoLog::class, 'carga_id', 'id');
    }
}

<?php

namespace App\Http\Requests;

use Illuminate\Foundation\Http\FormRequest;

class StoreCargaRequest extends FormRequest
{
    /**
     * Determina se o usuário está autorizado a fazer esta requisição.
     * Implementação da Arquitetura Zero Trust e Compliance.
     */
    public function authorize(): bool
    {
        $user = $this->user();

        // Só permite a requisição se for um embarcador com conta ativa.
        return $user !== null 
            && $user->role->slug === 'embarcador' 
            && $user->status === 'active';
    }

    /**
     * Regras de validação que serão aplicadas à requisição.
     */
    public function rules(): array
    {
        return [
            'produto' => 'required|string|max:255',
            // Alinhado com o frontend
            'especie' => 'required|string|in:caixas,paletes,sacaria,granel,tambores,outro',
            'peso_kg' => 'required|numeric|min:0.01',
            'cubagem_m3' => 'nullable|numeric|min:0.01',
            // Alinhado com o frontend (adicionado fiorino, carreta, vanderleia)
            'tipo_veiculo' => 'required|string|in:fiorino,toco,truck,bitruck,carreta,carreta_ls,vanderleia,bitrem',
            // Alinhado com o frontend ('aberta' no lugar de 'carga_seca')
            'tipo_carroceria' => 'required|string|in:bau,sider,aberta,graneleiro,frigorifico,prancha',
            'cidade_origem' => 'required|string|max:255',
            'uf_origem' => 'required|string|size:2',
            'cidade_destino' => 'required|string|max:255',
            'uf_destino' => 'required|string|size:2',
            'distancia_km' => 'nullable|numeric|min:1',
            'valor_frete' => 'required|numeric|min:1',
            'data_coleta' => 'required|date|after_or_equal:today',
            'data_entrega_prevista' => 'nullable|date|after_or_equal:data_coleta',
        ];
    }

    /**
     * Mensagens de erro personalizadas (opcional, melhora o feedback do frontend).
     */
    public function messages(): array
    {
        return [
            'tipo_veiculo.in' => 'O tipo de veículo selecionado é inválido.',
            'tipo_carroceria.in' => 'O tipo de carroceria selecionado é inválido.',
            'especie.in' => 'A espécie de embalagem selecionada é inválida.',
            'data_coleta.after_or_equal' => 'A data de coleta não pode estar no passado.',
            'data_entrega_prevista.after_or_equal' => 'A data de entrega não pode ser anterior à data de coleta.',
        ];
    }
}
