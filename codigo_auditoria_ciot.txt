namespace App\Http\Controllers\Api\V1\Embarcador;

use App\Http\Controllers\Controller;

class CargaController extends Controller
{
    /**
     * Publica uma nova carga no sistema (VIA FILA ASSÍNCRONA)
     */
    public function store(StoreCargaRequest $request)
    {
        $validated = $request->validated();
        $user = $request->user();

        // DISPARO ASSÍNCRONO PARA O REDIS
        ProcessarPublicacaoCarga::dispatch(
            $validated,
            $user->embarcador->id,
            $request->ip(),
            $request->header('User-Agent')
        )->onQueue('default');

        return response()->json([
            'message' => 'Lote recebido com sucesso. Sua carga está sendo processada e aparecerá no mural em instantes.',
        ], 202);
    }

    /**
     * Lista as cargas do Embarcador (Apenas as dele)
     */
    public function index(Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if (!$user->role || $user->role->slug !== 'embarcador' || !$user->embarcador) {
            return response()->json(['message' => 'Acesso restrito a embarcadores ativos.'], 403);
        }

        $cargas = Carga::with(['motorista.user:id,name,email', 'publicacao_log', 'aceite_log'])
            ->where('embarcador_id', $user->embarcador->id)
            ->orderBy('created_at', 'desc')
            ->paginate(15);

        return response()->json($cargas);
    }

    /**
     * Retorna os dados detalhados de uma carga específica
     */
    public function show(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para visualizar esta carga.'], 403);
        }

        // Embarcador agora carrega o CIOT associado para auditoria financeira
        $carga->load(['embarcador', 'motorista.user', 'aceite_log', 'publicacao_log', 'ciot']);

        return response()->json($carga);
    }

    /**
     * Atualiza os dados de uma carga existente
     */

    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para alterar esta carga.'], 403);
        }

        if ($carga->status !== 'publicada') {
            return response()->json(['message' => 'Cargas em negociação ou já alocadas não podem ser editadas. Cancele-a se necessário.'], 403);
        }

        $validated = $request->validated();
        
        // CORREÇÃO CRÍTICA: Arredondamento para evitar dízimas de ponto flutuante no banco de dados.

        ]);

    /**
     * Cancela (Exclusão Lógica) uma carga.
     */
    public function destroy(Carga $carga, Request $request)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado. Você não tem permissão para alterar esta carga.'], 403);
        }

        if ($carga->status !== 'publicada') {
            return response()->json(['message' => 'Esta carga não pode mais ser cancelada pois já está em negociação ou alocada.'], 403);
        }

        $carga->delete();

        return response()->json(['message' => 'Carga cancelada e removida do mural com sucesso.']);
    }

    /**
     * Aprova a auditoria de entrega (POD) e libera o pagamento do motorista.
     * Esta ação encerra a possibilidade de disputa.
     */
    public function aprovarEntrega(Request $request, Carga $carga, PefGatewayInterface $pefGateway)
    {
        $user = $request->user();
        $user->loadMissing('role', 'embarcador');

        if ($user->role && $user->role->slug === 'embarcador' && $carga->embarcador_id !== $user->embarcador->id) {
            return response()->json(['message' => 'Acesso negado.'], 403);
        }

        if ($carga->status !== 'em_auditoria') {
            return response()->json(['error' => 'Ação inválida. Esta carga não está aguardando aprovação de entrega.'], 400);
        }

        DB::transaction(function () use ($carga, $pefGateway) {
            $ciot = Ciot::where('carga_id', $carga->id)->first();
            
            if ($ciot && $ciot->status === 'emitido') {
                // Notifica a API da Pamcard/Mock para creditar o dinheiro no cartão do motorista
                $pefGateway->liquidarFrete($ciot->codigo_ciot);
                $ciot->update(['status' => 'liquidado']);
            }

            // A carga agora está oficialmente finalizada
            $carga->update(['status' => 'entregue']);
        });

        return response()->json([
            'message' => 'Auditoria aprovada com sucesso. O pagamento do motorista foi liberado na instituição financeira.',
            'carga' => $carga
        ], 200);
    }

namespace App\Jobs;

use App\Models\Carga;
use App\Models\User;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;
use Illuminate\Queue\InteractsWithQueue;
use Illuminate\Queue\SerializesModels;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;

class ProcessarAceiteCarga implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    protected $cargaId;
    protected $motoristaUserId;
    protected $motoristaPerfilId;
    protected $motoristaNome;
    protected $ipAddress;
    protected $userAgent;

    /**
     * Create a new job instance.
     */
    public function __construct(Carga $carga, User $user, $ipAddress, $userAgent)
    {
        $this->cargaId = $carga->id;
        $this->motoristaUserId = $user->id;
        $this->motoristaPerfilId = $user->motorista->id;
        $this->motoristaNome = $user->name;
        $this->ipAddress = $ipAddress;
        $this->userAgent = $userAgent;
    }

    /**
     * Execute the job.
     */
    public function handle(): void
    {
        // 1. Bloqueio Pessimista (Pessimistic Locking)
        // Isso impede que dois workers tentem dar update na mesma carga ao mesmo tempo
        DB::transaction(function () {
            $carga = Carga::where('id', $this->cargaId)->lockForUpdate()->first();

            // Verificação dupla: O worker valida se a carga AINDA está disponível no momento da execução
            if ($carga->status !== 'disponivel') {
                Log::info("Processamento cancelado: Carga {$this->cargaId} já alocada para outro motorista.");
                return; // Aborta silenciosamente
            }

            // 2. Geração do Contrato Imutável
            $termoContrato = "CONTRATO DE TRANSPORTE AUTÔNOMO DE CARGA. "
                           . "Pelo presente aceite eletrônico, o motorista {$this->motoristaNome} "
                           . "aceita realizar o transporte da carga ID {$carga->id}, "
                           . "com origem em {$carga->cidade_origem}/{$carga->uf_origem} e destino a {$carga->cidade_destino}/{$carga->uf_destino}, "
                           . "referente ao produto {$carga->produto} ({$carga->peso_kg}kg), "
                           . "pelo valor acordado de R$ " . number_format($carga->valor_frete, 2, ',', '.') . ". "
                           . "O motorista declara que possui CNH e RNTRC válidos, assume a responsabilidade civil sobre a mercadoria "
                           . "a partir do momento da coleta e isenta a plataforma intermediadora de qualquer vínculo empregatício "
                           . "ou responsabilidade sobre avarias durante o trajeto.";

            $termoHash = hash('sha256', $termoContrato);

            // 3. Execução das Queris
            $carga->update([
                'status' => 'alocada',
                'motorista_id' => $this->motoristaPerfilId
            ]);

            DB::table('carga_aceites_log')->insert([
                'carga_id' => $carga->id,
                'motorista_id' => $this->motoristaPerfilId,
                'ip_address' => $this->ipAddress,
                'user_agent' => $this->userAgent,
                'termo_hash' => $termoHash,
                'aceito_em' => now(),
                'created_at' => now(),
                'updated_at' => now(),
            ]);

            // FUTURO: Aqui entrará o Event() para disparar notificação WebSocket via Reverb
            Log::info("Carga {$carga->id} alocada com sucesso para o motorista {$this->motoristaPerfilId}");
        });
    }
}
